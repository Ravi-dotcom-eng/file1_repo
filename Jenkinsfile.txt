pipeline {
    agent any 
    tools {
       maven 'MAVEN'
        }
     
    stages{
    
         stage('Build') {
              steps {
                   sh 'mvn clean package'
                
              }
              post {
                 success{
                      echo "Archiving the Artifacts"
                      archiveArtifacts artifacts: '**/target/*.war'
                 }
              }
         }
         stage ('Deploy to tomcat server') {
           steps{
                deploy adapters: [tomcat9(path: '', url: 'http://54.187.198.1:8080/')], contextPath: null, war: '**/*.war'
              }
         }
     }
}
